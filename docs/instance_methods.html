_ = require "../lib/underscore.js"
_.mixin(require "../lib/inflection")
generateObjectId = require("../src/object_id")
socket = require("../lib/socket_singleton").getSocket()

module.exports = LiveDocumentInstanceMethods = 
  constructor: (@document) ->
    if !@document._id 
      @document._id = generateObjectId()
    @buildAssocs()

  get: (field) ->
    @document[field]

  buildAssocs: () ->
    _.each @constructor.hasOneAssocs, (assoc) =>
      query = {}
      query[@constructor.name + "_id"] = @_id
      @[assoc] =
        watch: (callback) =>
          @constructor.sendReadMessage _.pluralize(assoc), query, (docs) ->
            callback docs[0]   
        create: (document, callback) =>
          document[assoc + "_id"] = @_id
          @constructor.sendCreateMessage _.pluralize(assoc), document, (doc) ->
            callback doc  
        update: (callback) =>
          @constructor.sendUpdateMessage _.pluralize(assoc), query, document, (doc) ->
            callback doc
